=proto
void *duk_require_buffer(duk_context *ctx, int index, size_t *out_size);

=stack
[ ... val! ... ]

=summary
<p>Like <tt><a href="#duk_get_buffer">duk_get_buffer()</a></tt>,
but throws an error if the value at <tt>index</tt> is not a buffer
or if the index is invalid.</p>

=example
void *ptr;
size_t sz;

ptr = duk_require_buffer(ctx, &sz);
printf("buf=%p, size=%d\n", ptr, (int) sz);
