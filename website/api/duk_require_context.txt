=proto
duk_context *duk_require_context(duk_context *ctx, int index);

=stack
[ ... val! ... ]

=summary
<p>Like <tt><a href="#duk_get_context">duk_get_context()</a></tt>,
but throws an error if the value at <tt>index</tt> is not a Duktape
thread or if the index is invalid.</p>

=example
duk_context *new_ctx;

(void) duk_push_new_thread(ctx);
new_ctx = duk_require_context(ctx, -1);
