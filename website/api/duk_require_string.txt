=proto
const char *duk_require_string(duk_context *ctx, int index);

=stack
[ ... val! ... ]

=summary
<p>Like <tt><a href="#duk_get_string">duk_get_string()</a></tt>,
but throws an error if the value at <tt>index</tt> is not a string
or if the index is invalid.</p>

=example
const char *buf;

buf = duk_require_string(ctx);
printf("value is a string: %s\n", buf);
